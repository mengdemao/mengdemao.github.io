简介

ANTLR是一款强大的语法分析器生成工具,用于读取、处理、执行和翻译结构化的文本或二进制文件.

类似于flex/bison,根据描述文件，自动生成词法语法分析器

antlr语法详解

Hello

1// antlr4 Hello.g4
2// javac *.java
3// grun Hello r -gui
4grammar Hello;              // 定义一个Hello的grammer
5r  : 'hello' ID ;           // 开头是hello后面接着一个ID
6ID : [a-z]+ ;               // ID由小写字母组成
7WS : [ \t\r\n]+ -> skip ;   // 控制符清除

ArrayInit

 1// antlr4 ArrayInit.g4
 2// javac *.java
 3// grun ArrayInit init -gui
 4grammar ArrayInit;						// 定义一个ArrayInit的grammer
 5init    : '{' value (',' value)* '}' ; 	// 
 6value   : init							// 嵌套定义
 7        | INT							// 定义整数
 8        ;
 9INT     :   [0-9]+ ;
10WS      :   [ \t\r\n]+ -> skip ;

()* –> 相当于扩展

Expr

 1// antlr4 Expr.g4
 2// javac *.java
 3// grun Expr prog -gui
 4grammar Expr;
 5
 6prog : stat+;
 7
 8stat: expr NEWLINE          # printExpr
 9    | ID '=' expr NEWLINE   # assign
10    | NEWLINE               # blank
11    ;
12
13expr: expr op=('*'|'/') expr        # MulDiv
14    | expr op=('+'|'-') expr        # AddSub
15    | INT                           # int
16    | ID                            # id
17    | '('expr')'                  # parens
18    ;
19
20MUL     : '*' ; // assigns token name to '*' used above in grammar
21DIV     : '/' ;
22ADD     : '+' ;
23SUB     : '-' ;
24ID      : [a-zA-Z]+ ;
25INT     : [0-9]+ ;
26NEWLINE :'\r'? '\n' ;
27WS      : [ \t]+ -> skip;

json

在词法规则中那些不会被语法规则直接调用的词法规则可以用一个fragment关键字来标识， fragment标识的规则只能为其它词法规则提供基础

 1grammar JSON;	// 声明一个grammar
 2
 3json
 4   : value		// 一个value候选
 5   ;
 6
 7obj				// 对象类型
 8   : '{' pair (',' pair)* '}'
 9   | '{' '}'
10   ;
11
12pair
13   : STRING ':' value
14   ;
15
16arr
17   : '[' value (',' value)* ']'
18   | '[' ']'
19   ;
20
21value
22   : STRING
23   | NUMBER
24   | obj
25   | arr
26   | 'true'
27   | 'false'
28   | 'null'
29   ;
30
31
32STRING
33   : '"' (ESC | SAFECODEPOINT)* '"'
34   ;
35
36
37fragment ESC
38   : '\\' (["\\/bfnrt] | UNICODE)
39   ;
40
41
42fragment UNICODE
43   : 'u' HEX HEX HEX HEX
44   ;
45
46
47fragment HEX
48   : [0-9a-fA-F]
49   ;
50
51
52fragment SAFECODEPOINT
53   : ~ ["\\\u0000-\u001F]
54   ;
55
56
57NUMBER
58   : '-'? INT ('.' [0-9] +)? EXP?
59   ;
60
61
62fragment INT
63   : '0' | [1-9] [0-9]*
64   ;
65
66// no leading zeros
67
68fragment EXP
69   : [Ee] [+\-]? INT
70   ;
71
72// \- since - means "range" inside [...]
73
74WS
75   : [ \t\n\r] + -> skip
76   ;

  1grammar DOT;
  2
  3graph
  4   : STRICT? ( GRAPH | DIGRAPH ) id_? '{' stmt_list '}'
  5   ;
  6
  7stmt_list
  8   : ( stmt ';'? )*
  9   ;
 10
 11stmt
 12   : node_stmt | edge_stmt | attr_stmt | id_ '=' id_ | subgraph
 13   ;
 14
 15attr_stmt
 16   : ( GRAPH | NODE | EDGE ) attr_list
 17   ;
 18
 19attr_list
 20   : ( '[' a_list? ']' )+
 21   ;
 22
 23a_list
 24   : ( id_ ( '=' id_ )? ','? )+
 25   ;
 26
 27edge_stmt
 28   : ( node_id | subgraph ) edgeRHS attr_list?
 29   ;
 30
 31edgeRHS
 32   : ( edgeop ( node_id | subgraph ) )+
 33   ;
 34
 35edgeop
 36   : '->' | '--'
 37   ;
 38
 39node_stmt
 40   : node_id attr_list?
 41   ;
 42
 43node_id
 44   : id_ port?
 45   ;
 46
 47port
 48   : ':' id_ ( ':' id_ )?
 49   ;
 50
 51subgraph
 52   : ( SUBGRAPH id_? )? '{' stmt_list '}'
 53   ;
 54
 55id_
 56   : ID | STRING | HTML_STRING | NUMBER
 57   ;
 58
 59// "The keywords node, edge, graph, digraph, subgraph, and strict are
 60// case-independent"
 61
 62STRICT
 63   : [Ss] [Tt] [Rr] [Ii] [Cc] [Tt]
 64   ;
 65
 66
 67GRAPH
 68   : [Gg] [Rr] [Aa] [Pp] [Hh]
 69   ;
 70
 71
 72DIGRAPH
 73   : [Dd] [Ii] [Gg] [Rr] [Aa] [Pp] [Hh]
 74   ;
 75
 76
 77NODE
 78   : [Nn] [Oo] [Dd] [Ee]
 79   ;
 80
 81
 82EDGE
 83   : [Ee] [Dd] [Gg] [Ee]
 84   ;
 85
 86
 87SUBGRAPH
 88   : [Ss] [Uu] [Bb] [Gg] [Rr] [Aa] [Pp] [Hh]
 89   ;
 90
 91
 92/** "a numeral [-]?(.[0-9]+ | [0-9]+(.[0-9]*)? )" */ NUMBER
 93   : '-'? ( '.' DIGIT+ | DIGIT+ ( '.' DIGIT* )? )
 94   ;
 95
 96
 97fragment DIGIT
 98   : [0-9]
 99   ;
100
101
102/** "any double-quoted string ("...") possibly containing escaped quotes" */ STRING
103   : '"' ( '\\"' | . )*? '"'
104   ;
105
106
107/** "Any string of alphabetic ([a-zA-Z\200-\377]) characters, underscores
108 *  ('_') or digits ([0-9]), not beginning with a digit"
109 */ ID
110   : LETTER ( LETTER | DIGIT )*
111   ;
112
113
114fragment LETTER
115   : [a-zA-Z\u0080-\u00FF_]
116   ;
117
118
119/** "HTML strings, angle brackets must occur in matched pairs, and
120 *  unescaped newlines are allowed."
121 */ HTML_STRING
122   : '<' ( TAG | ~ [<>] )* '>'
123   ;
124
125
126fragment TAG
127   : '<' .*? '>'
128   ;
129
130
131COMMENT
132   : '/*' .*? '*/' -> skip
133   ;
134
135
136LINE_COMMENT
137   : '//' .*? '\r'? '\n' -> skip
138   ;
139
140
141/** "a '#' character is considered a line output from a C preprocessor (e.g.,
142 *  # 34 to indicate line 34 ) and discarded"
143 */ PREPROC
144   : '#' ~[\r\n]* -> skip
145   ;
146
147
148WS

150   ;
