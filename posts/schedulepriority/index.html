调度优先级

Nov 14, 2021 3 分钟阅读

优先级定义

内核优先级使用0-139确定内部优先级,0-99分配实时进程; [100, 139]分配非实时进程,nice的映射[-20, 19]与非实时优先级的相对应.

计算静态优先级

 1static int effective_prio(struct task_struct *p)
 2{
 3	/* 设置普通优先级 */
 4	p->normal_prio = normal_prio(p);
 5
 6	/* 如果不是实时优先级,则返回普通优先级 */
 7	if (!rt_prio(p->prio))
 8		return p->normal_prio;
 9	
10	/* 返回动态优先级 */
11	return p->prio;
12}

获取普通优先级

 1static inline int __normal_prio(struct task_struct *p)
 2{
 3	return p->static_prio;
 4}
 5
 6static inline int normal_prio(struct task_struct *p)
 7{
 8	int prio;
 9
10	if (task_has_dl_policy(p))
11		prio = MAX_DL_PRIO-1;
12	else if (task_has_rt_policy(p))
13		prio = MAX_RT_PRIO-1 - p->rt_priority;
14	else
15		prio = __normal_prio(p);
16	return prio;
17}

 30struct sched_dl_entity {
 31	struct rb_node			rb_node;
 32
 33	/*
 34	 * Original scheduling parameters. Copied here from sched_attr
 35	 * during sched_setattr(), they will remain the same until
 36	 * the next sched_setattr().
 37	 */
 38	u64				dl_runtime;	/* Maximum runtime for each instance	*/
 39	u64				dl_deadline;	/* Relative deadline of each instance	*/
 40	u64				dl_period;	/* Separation of two instances (period) */
 41	u64				dl_bw;		/* dl_runtime / dl_period		*/
 42	u64				dl_density;	/* dl_runtime / dl_deadline		*/
 43
 44	/*
 45	 * Actual scheduling parameters. Initialized with the values above,
 46	 * they are continously updated during task execution. Note that
 47	 * the remaining runtime could be < 0 in case we are in overrun.
 48	 */
 49	s64				runtime;	/* Remaining runtime for this instance	*/
 50	u64				deadline;	/* Absolute deadline for this instance	*/
 51	unsigned int			flags;		/* Specifying the scheduler behaviour	*/
 52
 53	/*
 54	 * Some bool flags:
 55	 *
 56	 * @dl_throttled tells if we exhausted the runtime. If so, the
 57	 * task has to wait for a replenishment to be performed at the
 58	 * next firing of dl_timer.
 59	 *
 60	 * @dl_boosted tells if we are boosted due to DI. If so we are
 61	 * outside bandwidth enforcement mechanism (but only until we
 62	 * exit the critical section);
 63	 *
 64	 * @dl_yielded tells if task gave up the CPU before consuming
 65	 * all its available runtime during the last job.
 66	 *
 67	 * @dl_non_contending tells if the task is inactive while still
 68	 * contributing to the active utilization. In other words, it
 69	 * indicates if the inactive timer has been armed and its handler
 70	 * has not been executed yet. This flag is useful to avoid race
 71	 * conditions between the inactive timer handler and the wakeup
 72	 * code.
 73	 *
 74	 * @dl_overrun tells if the task asked to be informed about runtime
 75	 * overruns.
 76	 */
 77	unsigned int			dl_throttled      : 1;
 78	unsigned int			dl_boosted        : 1;
 79	unsigned int			dl_yielded        : 1;
 80	unsigned int			dl_non_contending : 1;
 81	unsigned int			dl_overrun	  : 1;
 82
 83	/*
 84	 * Bandwidth enforcement timer. Each -deadline task has its
 85	 * own bandwidth to be enforced, thus we need one timer per task.
 86	 */
 87	struct hrtimer			dl_timer;
 88
 89	/*
 90	 * Inactive timer, responsible for decreasing the active utilization
 91	 * at the "0-lag time". When a -deadline task blocks, it contributes
 92	 * to GRUB's active utilization until the "0-lag time", hence a
 93	 * timer is needed to decrease the active utilization at the correct
 94	 * time.
 95	 */
 96	struct hrtimer inactive_timer;
 97};

23static void set_load_weight(struct task_struct *p, bool update_load)
24{
25	int prio = p->static_prio - MAX_RT_PRIO;
26	struct load_weight *load = &p->se.load;
27
28	// SCHED_IDLE 拥有最低权重
29	if (idle_policy(p->policy)) {
30		load->weight     = scale_load(WEIGHT_IDLEPRIO);
31		load->inv_weight = WMULT_IDLEPRIO;
32		return;
33	}
34
35	if (update_load && p->sched_class == &fair_sched_class) {
36		reweight_task(p, prio);
37	} else {
38		load->weight = scale_load(sched_prio_to_weight[prio]);
39		load->inv_weight = sched_prio_to_wmult[prio];
40	}
41}

需要更新权重

 1void reweight_task(struct task_struct *p, int prio)
 2{
 3	struct sched_entity *se = &p->se;
 4	struct cfs_rq *cfs_rq = cfs_rq_of(se);
 5	struct load_weight *load = &se->load;
 6	unsigned long weight = scale_load(sched_prio_to_weight[prio]);
 7
 8	reweight_entity(cfs_rq, se, weight, weight);
 9	load->inv_weight = sched_prio_to_wmult[prio];
10}
11static void reweight_entity(struct cfs_rq *cfs_rq, 
12							struct sched_entity *se,
13							unsigned long weight, 
14							unsigned long runnable)
15{
16	if (se->on_rq) {
17		/* commit outstanding execution time */
18		if (cfs_rq->curr == se)
19			update_curr(cfs_rq);
20		account_entity_dequeue(cfs_rq, se);
21		dequeue_runnable_load_avg(cfs_rq, se);
22	}
23	dequeue_load_avg(cfs_rq, se);
24
25	se->runnable_weight = runnable;
26	update_load_set(&se->load, weight);
27	enqueue_load_avg(cfs_rq, se);
28	if (se->on_rq) {
29		account_entity_enqueue(cfs_rq, se);
30		enqueue_runnable_load_avg(cfs_rq, se);
31	}
32}
