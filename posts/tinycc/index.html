TinyCC

TinyCC (aka TCC) is a small but hyper fast C compiler. Unlike other C compilers, it is meant to be self-relying: you do not need an external assembler or linker because TCC does that for you.

 1LIBTCCAPI TCCState *tcc_new(void)
 2{
 3    TCCState *s;
 4
 5    s = tcc_mallocz(sizeof(TCCState));
 6    if (!s)
 7        return NULL;
 8#ifdef MEM_DEBUG
 9    ++nb_states;
10#endif
11
12#undef gnu_ext
13
14    s->gnu_ext = 1;
15    s->tcc_ext = 1;
16    s->nocommon = 1;
17    s->dollars_in_identifiers = 1; /*on by default like in gcc/clang*/
18    s->cversion = 199901; /* default unless -std=c11 is supplied */
19    s->warn_implicit_function_declaration = 1;
20    s->warn_discarded_qualifiers = 1;
21    s->ms_extensions = 1;
22
23#ifdef CHAR_IS_UNSIGNED
24    s->char_is_unsigned = 1;
25#endif
26#ifdef TCC_TARGET_I386
27    s->seg_size = 32;
28#endif
29    /* enable this if you want symbols with leading underscore on windows: */
30#if defined TCC_TARGET_MACHO /* || defined TCC_TARGET_PE */
31    s->leading_underscore = 1;
32#endif
33#ifdef TCC_TARGET_ARM
34    s->float_abi = ARM_FLOAT_ABI;
35#endif
36
37    s->ppfp = stdout;
38    /* might be used in error() before preprocess_start() */
39    s->include_stack_ptr = s->include_stack;
40
41    tccelf_new(s);
42
43    tcc_set_lib_path(s, CONFIG_TCCDIR);
44    return s;
45}

47LIBTCCAPI void tcc_delete(TCCState *s1)
48{
49    /* free sections */
50    tccelf_delete(s1);
51
52    /* free library paths */
53    dynarray_reset(&s1->library_paths, &s1->nb_library_paths);
54    dynarray_reset(&s1->crt_paths, &s1->nb_crt_paths);
55
56    /* free include paths */
57    dynarray_reset(&s1->include_paths, &s1->nb_include_paths);
58    dynarray_reset(&s1->sysinclude_paths, &s1->nb_sysinclude_paths);
59
60    tcc_free(s1->tcc_lib_path);
61    tcc_free(s1->soname);
62    tcc_free(s1->rpath);
63    tcc_free(s1->init_symbol);
64    tcc_free(s1->fini_symbol);
65    tcc_free(s1->outfile);
66    tcc_free(s1->deps_outfile);
67    dynarray_reset(&s1->files, &s1->nb_files);
68    dynarray_reset(&s1->target_deps, &s1->nb_target_deps);
69    dynarray_reset(&s1->pragma_libs, &s1->nb_pragma_libs);
70    dynarray_reset(&s1->argv, &s1->argc);
71    cstr_free(&s1->cmdline_defs);
72    cstr_free(&s1->cmdline_incl);
73#ifdef TCC_IS_NATIVE
74    /* free runtime memory */
75    tcc_run_free(s1);
76#endif
77
78    tcc_free(s1);
79#ifdef MEM_DEBUG
80    if (0 == --nb_states)
81        tcc_memcheck();
82#endif
83}

 1static int _tcc_open(TCCState *s1, const char *filename)
 2{
 3    int fd;
 4    if (strcmp(filename, "-") == 0)
 5        fd = 0, filename = "<stdin>";
 6    else
 7        fd = open(filename, O_RDONLY | O_BINARY);
 8    if ((s1->verbose == 2 && fd >= 0) || s1->verbose == 3)
 9        printf("%s %*s%s\n", fd < 0 ? "nf":"->",
10               (int)(s1->include_stack_ptr - s1->include_stack), "", filename);
11    return fd;
12}
13
14ST_FUNC int tcc_open(TCCState *s1, const char *filename)
15{
16    int fd = _tcc_open(s1, filename);
17    if (fd < 0)
18        return -1;
19    tcc_open_bf(s1, filename, 0);
20    file->fd = fd;
21    return 0;
22}
23
24ST_FUNC void tcc_close(void)
25{
26    TCCState *s1 = tcc_state;
27    BufferedFile *bf = file;
28    if (bf->fd > 0) {
29        close(bf->fd);
30        total_lines += bf->line_num;
31    }
32    if (bf->true_filename != bf->filename)
33        tcc_free(bf->true_filename);
34    file = bf->prev;
35    tcc_free(bf);
36}
