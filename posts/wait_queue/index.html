例子

写端

 1ssize_t wait_write(struct file *file, const char __user *data, size_t len, loff_t *ppos)
 2{
 3	DECLARE_WAITQUEUE(wait, current);		/* 声明等待队列 */
 4	int ret = -1;
 5	PTRACE;
 6
 7	mutex_lock(&wait_device.mutex);
 8	/* 非阻塞模式直接写入 */
 9	if (file->f_flags & O_NONBLOCK) {
10		pr_err("write in O_NONBLOCK Mode");
11		goto pure_write;
12	}
13
14	add_wait_queue(&wait_device.wait_w, &wait);
15	while (wait_device.wait_flag == true) {
16		pr_err("Write INTERRUPTIBLE");
17		__set_current_state(TASK_INTERRUPTIBLE);
18		mutex_unlock(&wait_device.mutex);
19		schedule();
20		if (signal_pending(current)) {
21			ret = -ERESTARTSYS;
22			remove_wait_queue(&wait_device.wait_w, &wait);
23			__set_current_state(TASK_RUNNING);
24			goto out;
25		}
26	}
27	remove_wait_queue(&wait_device.wait_w, &wait);
28
29pure_write:
30	wait_device.wait_flag = true;
31	pr_err("Write Successful");
32
33	wake_up_interruptible(&wait_device.wait_r);
34	pr_err("Wakeup Read");
35	goto out;
36
37out:
38	mutex_unlock(&wait_device.mutex);
39	return ret;
40}
