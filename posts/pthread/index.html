Linux系统调用

用户层调用SVC指令实现

 1#define __asm_syscall(...) 
 2do {
 3	__asm__ __volatile__ ( "svc 0" : "=r"(r0) : __VA_ARGS__ : "memory");
 4	return r0;
 5} while (0);
 6
 7#define R7_OPERAND "r"(r7)
 8static inline long __syscall0(long n)
 9{
10	register long r7 __ASM____R7__ = n; //使用R7传递个数
11	register long r0 __asm__("r0");
12	__asm_syscall(R7_OPERAND);
13}
14
15static inline long __syscall1(long n, long a)
16{
17	register long r7 __ASM____R7__ = n;
18	register long r0 __asm__("r0") = a;
19	__asm_syscall(R7_OPERAND, "0"(r0));
20}
21
22static inline long __syscall6(long n, long a, long b, long c, long d, long e, long f)
23{
24	register long r7 __ASM____R7__ = n;
25	register long r0 __asm__("r0") = a;
26	register long r1 __asm__("r1") = b;
27	register long r2 __asm__("r2") = c;
28	register long r3 __asm__("r3") = d;
29	register long r4 __asm__("r4") = e;
30	register long r5 __asm__("r5") = f;
31	__asm_syscall(R7_OPERAND, "0"(r0), "r"(r1), "r"(r2), "r"(r3), "r"(r4), "r"(r5));
32}
33
34// 一个相当巧妙的宏定义的实现
35#define __SYSCALL_NARGS_X(a,b,c,d,e,f,g,h,n,...) n
36#define __SYSCALL_NARGS(...) __SYSCALL_NARGS_X(__VA_ARGS__,7,6,5,4,3,2,1,0,)
37// 每次添加一个参数,就会将数字向后面推一个位置,形成参数个数
38
39// 拼装函数调用
40#define __SYSCALL_CONCAT_X(a,b) a##b
41#define __SYSCALL_CONCAT(a,b) __SYSCALL_CONCAT_X(a,b)
42#define __SYSCALL_DISP(b,...) __SYSCALL_CONCAT(b,__SYSCALL_NARGS(__VA_ARGS__))(__VA_ARGS__)
43
44// __syscall_ret仅仅检查了系统调用号
45#define __syscall(...) __SYSCALL_DISP(__syscall,__VA_ARGS__)
46#define syscall(...) __syscall_ret(__syscall(__VA_ARGS__))
  1. syscall(SYS_close, fd) 原始函数
  2. __syscall_ret(__syscall(SYS_close, fd)) syscall宏定义展开
  3. __syscall(SYS_close, fd) 直接拿掉syscall_ret
  4. __SYSCALL_DISP(__syscall, SYS_close, fd) 展开__syscall
  5. __SYSCALL_CONCAT(__syscall, __SYSCALL_NARGS(SYS_close, fd))(SYS_close, fd)
  6. __SYSCALL_CONCAT(_syscall, 1, (SYS_close, fd))
  7. __syscall1(SYS_close, fd) 生成结束
