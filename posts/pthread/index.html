但是在我们使用的时候发现与我们正常的使用不太一致,在用户层,我们一般认为pthread为线程ID，但是内部实现好像都是指针,因此出现了什么特殊的原因；

在musl中,直接抹掉了内部结构;

1// 因此在此处使用了技巧,在内部和外部使用的定义形式不一致

 1long do_futex(u32 __user *uaddr, int op, u32 val, ktime_t *timeout,
 2		u32 __user *uaddr2, u32 val2, u32 val3)
 3{
 4	int cmd = op & FUTEX_CMD_MASK;
 5	unsigned int flags = 0;
 6
 7	if (!(op & FUTEX_PRIVATE_FLAG))
 8		flags |= FLAGS_SHARED;
 9
10	if (op & FUTEX_CLOCK_REALTIME) {
11		flags |= FLAGS_CLOCKRT;
12		if (cmd != FUTEX_WAIT && cmd != FUTEX_WAIT_BITSET && \
13		    cmd != FUTEX_WAIT_REQUEUE_PI)
14			return -ENOSYS;
15	}
16
17	switch (cmd) {
18	case FUTEX_LOCK_PI:
19	case FUTEX_UNLOCK_PI:
20	case FUTEX_TRYLOCK_PI:
21	case FUTEX_WAIT_REQUEUE_PI:
22	case FUTEX_CMP_REQUEUE_PI:
23		if (!futex_cmpxchg_enabled)
24			return -ENOSYS;
25	}
26
27	switch (cmd) {
28	case FUTEX_WAIT:
29		val3 = FUTEX_BITSET_MATCH_ANY;
30		/* fall through */
31	case FUTEX_WAIT_BITSET:
32		return futex_wait(uaddr, flags, val, timeout, val3);
33	case FUTEX_WAKE:
34		val3 = FUTEX_BITSET_MATCH_ANY;
35		/* fall through */
36	case FUTEX_WAKE_BITSET:
37		return futex_wake(uaddr, flags, val, val3);
38	case FUTEX_REQUEUE:
39		return futex_requeue(uaddr, flags, uaddr2, val, val2, NULL, 0);
40	case FUTEX_CMP_REQUEUE:
41		return futex_requeue(uaddr, flags, uaddr2, val, val2, &val3, 0);
42	case FUTEX_WAKE_OP:
43		return futex_wake_op(uaddr, flags, uaddr2, val, val2, val3);
44	case FUTEX_LOCK_PI:
45		return futex_lock_pi(uaddr, flags, timeout, 0);
46	case FUTEX_UNLOCK_PI:
47		return futex_unlock_pi(uaddr, flags);
48	case FUTEX_TRYLOCK_PI:
49		return futex_lock_pi(uaddr, flags, NULL, 1);
50	case FUTEX_WAIT_REQUEUE_PI:
51		val3 = FUTEX_BITSET_MATCH_ANY;
52		return futex_wait_requeue_pi(uaddr, flags, val, timeout, val3,
53					     uaddr2);
54	case FUTEX_CMP_REQUEUE_PI:
55		return futex_requeue(uaddr, flags, uaddr2, val, val2, &val3, 1);
56	}
57	return -ENOSYS;

 41/**

 43 * @brief 创建线程
 44 *
 45 * @param res 		pthread指针
 46 * @param attrp		pthread属性
 47 * @param entry		入口地址
 48 * @param arg		入口属性
 49 * @return			创建结果

 51int pthread_create(pthread_t *restrict res,
 52				   const pthread_attr_t *restrict attrp,
 53				   void *(*entry)(void *), 
 54                   void *restrict arg)
 55{
 56    int ret, c11 = (attrp == __ATTRP_C11_THREAD);
 57	size_t size;
 58    size_t guard;
 59	struct pthread *self;
 60    struct pthread *new;
 61	unsigned char *map = 0, *stack = 0, *tsd = 0, *stack_limit;
 62	unsigned flags = CLONE_VM | CLONE_FS | CLONE_FILES | CLONE_SIGHAND
 63		| CLONE_THREAD | CLONE_SYSVSEM | CLONE_SETTLS
 64		| CLONE_PARENT_SETTID | CLONE_CHILD_CLEARTID | CLONE_DETACHED;
 65	
 66    // 通过一定的规则将attrp-->attr
 67    pthread_attr_t attr = { 0 };
 68	sigset_t set;
 69    
 70    /* 如果没有设置堆栈大小,先设置默认的堆栈大小 */
 71    attr._a_stacksize = __default_stacksize;
 72	attr._a_guardsize = __default_guardsize;
 73    
 74    /* 填充pthread成员 */
 75    new = __copy_tls(tsd - libc.tls_size);
 76	new->map_base = map;
 77	new->map_size = size;
 78	new->stack = stack;
 79	new->stack_size = stack - stack_limit;
 80	new->guard_size = guard;
 81	new->self = new;
 82	new->tsd = (void *)tsd;
 83	new->locale = &libc.global_locale;
 84	if (attr._a_detach) {
 85		new->detach_state = DT_DETACHED;
 86	} else {
 87		new->detach_state = DT_JOINABLE;
 88	}
 89	new->robust_list.head = &new->robust_list.head;
 90	new->canary = self->canary;
 91	new->sysinfo = self->sysinfo;
 92    
 93    /* 移动stack指针,保存启动参数 */
 94    stack -= (uintptr_t)stack % sizeof(uintptr_t);
 95	stack -= sizeof(struct start_args);
 96	
 97    /* 设置启动参数 */
 98    struct start_args *args = (void *)stack;
 99	args->start_func = entry;
100	args->start_arg = arg;
101	args->control = attr._a_sched ? 1 : 0;
102    
103    /* 调用clone创建线程  */
104    clone(start, stack, flags, args, &new->tid, TP_ADJ(new), &__thread_list_lock);
105	
106    /* 设置调度器 */
107    ret = __syscall(SYS_sched_setscheduler, new->tid, attr._a_policy, &attr._a_prio);
108	if (a_swap(&args->control, ret ? 3 : 0)==2)
109		__wake(&args->control, 1, 1);
110	if (ret)
111		__wait(&args->control, 0, 3, 0);
112    
113    /* 初始化链表 */
114    new->next = self->next;
115	new->prev = self;
116	new->next->prev = new;
117	new->prev->next = new;
118    
119    /* 返回new作为线程ID */
120}

122void __pthread_exit(void *result)
123{
124	pthread_t self = __pthread_self();
125	sigset_t set;
126    
127    /*  设置退出标志 */
128    self->canceldisable = 1;
129	self->cancelasync = 0;
130	self->result = result;
131	
132    /* 执行线程清理函数 */
133	while (self->cancelbuf) {
134		void (*f)(void *) = self->cancelbuf->__f;
135		void *x = self->cancelbuf->__x;
136		self->cancelbuf = self->cancelbuf->__next;
137		f(x);
138	}
139
140    int state = a_cas(&self->detach_state, DT_JOINABLE, DT_EXITING);
141	
142    /* 如果线程分离状态,则代表需要自己手动释放内存 */
143	if (state==DT_DETACHED && self->map_base) {
144        __vm_wait();
145	}
146    
147    volatile void *volatile *rp;
148	while ((rp=self->robust_list.head) && rp != &self->robust_list.head) {
149		pthread_mutex_t *m = (void *)((char *)rp
150			- offsetof(pthread_mutex_t, _m_next));
151		int waiters = m->_m_waiters;
152		int priv = (m->_m_type & 128) ^ 128;
153		self->robust_list.pending = rp;
154		self->robust_list.head = *rp;
155		int cont = a_swap(&m->_m_lock, 0x40000000);
156		self->robust_list.pending = 0;
157		if (cont < 0 || waiters)
158			__wake(&m->_m_lock, 1, priv);
159	}
160
161    self->next->prev = self->prev;
162	self->prev->next = self->next;
163	self->prev = self->next = self;
164
165	if (state==DT_DETACHED && self->map_base) {
166		if (self->robust_list.off) {
167			__syscall(SYS_set_robust_list, 0, 3*sizeof(long));
168        }
169        
170		__unmapself(self->map_base, self->map_size);
171	}
172    
173    /* 设置线程为退出 */
174    a_store(&self->detach_state, DT_EXITED);
175	
176    /* 唤醒 */
177    __wake(&self->detach_state, 1, 1);
178
179    /* 清除线程ID  */
180	self->tid = 0;
181
182    /* 杀死线程 */
183	for (;;) __syscall(SYS_exit, 0);
184}

72		if (r && r != EINTR) break;
73	}
74	return r;
75}

上锁的实现

 1int __pthread_mutex_lock (pthread_mutex_t *mtxp)
 2{
 3	struct __pthread *self;
 4	int flags = mtxp->__flags & GSYNC_SHARED;
 5	int ret = 0;
 6  
 7	// 根据不同的类型确定
 8	#define PT_MTX_NORMAL       __PTHREAD_MUTEX_TIMED
 9	#define PT_MTX_RECURSIVE    (__PTHREAD_MUTEX_RECURSIVE + 1)
10	#define PT_MTX_ERRORCHECK   (__PTHREAD_MUTEX_ERRORCHECK + 1)
11	switch (MTX_TYPE (mtxp))
12	{
13	/* 普通上锁方式 */
14	case PT_MTX_NORMAL:
15		lll_lock (mtxp->__lock, flags);
16		break;
17	
18	/* 迭代锁 */
19	case PT_MTX_RECURSIVE:
20		self = _pthread_self ();
21		if (mtx_owned_p (mtxp, self, flags))
22		{
23			if (__glibc_unlikely (mtxp->__cnt + 1 == 0))
24			return EAGAIN;
25
26			++mtxp->__cnt;
27			return ret;
28		}
29
30		lll_lock (mtxp->__lock, flags);
31		mtx_set_owner (mtxp, self, flags);
32		mtxp->__cnt = 1;
33		break;
34
35	case PT_MTX_ERRORCHECK:
36		self = _pthread_self ();
37		if (mtx_owned_p (mtxp, self, flags))
38			return EDEADLK;
39
40		lll_lock (mtxp->__lock, flags);
41		mtx_set_owner (mtxp, self, flags);
42		break;
43
44	case PT_MTX_NORMAL | PTHREAD_MUTEX_ROBUST:
45	case PT_MTX_RECURSIVE | PTHREAD_MUTEX_ROBUST:
46	case PT_MTX_ERRORCHECK | PTHREAD_MUTEX_ROBUST:
47		self = _pthread_self ();
48		ROBUST_LOCK (self, mtxp, lll_robust_lock, flags);
49		break;
50
51	default:
52		ret = EINVAL;
53		break;
54	}
55
56  return ret;
57}

普通上锁实现原理

 1#define __lll_lock(futex, private)                                      \
 2  ((void)                                                               \
 3   ({                                                                   \
 4     int *__futex = (futex);                                            \
 5     if (__glibc_unlikely                                               \
 6         (atomic_compare_and_exchange_bool_acq (__futex, 1, 0)))        \
 7       {                                                                \
 8         if (__builtin_constant_p (private) && (private) == LLL_PRIVATE) \
 9           __lll_lock_wait_private (__futex);                           \
10         else                                                           \
11           __lll_lock_wait (__futex, private);                          \
12       }                                                                \
13   }))
14#define lll_lock(futex, private) 			__lll_lock (&(futex), private)
15
16void __lll_lock_wait_private (int *futex)
17{
18  if (atomic_load_relaxed (futex) == 2)
19    goto futex;
20
21  while (atomic_exchange_acquire (futex, 2) != 0)
22    {
23    futex:
24      LIBC_PROBE (lll_lock_wait_private, 1, futex);
25      futex_wait ((unsigned int *) futex, 2, LLL_PRIVATE); /* Wait if *futex == 2.  */
26    }
27}
28
29void __lll_lock_wait (int *futex, int private)
30{
31  if (atomic_load_relaxed (futex) == 2)
32    goto futex;
33
34  while (atomic_exchange_acquire (futex, 2) != 0)
35    {
36    futex:
37      LIBC_PROBE (lll_lock_wait, 1, futex);
38      futex_wait((unsigned int *) futex, 2, private); /* Wait if *futex == 2.  */
39    }
40}
