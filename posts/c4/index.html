stack based virtual machine

 1  while (1) {
 2    i = *pc++; ++cycle;
 3    if      (i == LEA) a = (int)(bp + *pc++);                             // load local address
 4    else if (i == IMM) a = *pc++;                                         // load global address or immediate
 5    else if (i == JMP) pc = (int *)*pc;                                   // jump
 6    else if (i == JSR) { *--sp = (int)(pc + 1); pc = (int *)*pc; }        // jump to subroutine
 7    else if (i == BZ)  pc = a ? pc + 1 : (int *)*pc;                      // branch if zero
 8    else if (i == BNZ) pc = a ? (int *)*pc : pc + 1;                      // branch if not zero
 9    else if (i == ENT) { *--sp = (int)bp; bp = sp; sp = sp - *pc++; }     // enter subroutine
10    else if (i == ADJ) sp = sp + *pc++;                                   // stack adjust
11    else if (i == LEV) { sp = bp; bp = (int *)*sp++; pc = (int *)*sp++; } // leave subroutine
12    else if (i == LI)  a = *(int *)a;                                     // load int
13    else if (i == LC)  a = *(char *)a;                                    // load char
14    else if (i == SI)  *(int *)*sp++ = a;                                 // store int
15    else if (i == SC)  a = *(char *)*sp++ = a;                            // store char
16    else if (i == PSH) *--sp = a;                                         // push
17
18    else if (i == OR)  a = *sp++ |  a;
19    else if (i == XOR) a = *sp++ ^  a;
20    else if (i == AND) a = *sp++ &  a;
21    else if (i == EQ)  a = *sp++ == a;
22    else if (i == NE)  a = *sp++ != a;
23    else if (i == LT)  a = *sp++ <  a;
24    else if (i == GT)  a = *sp++ >  a;
25    else if (i == LE)  a = *sp++ <= a;
26    else if (i == GE)  a = *sp++ >= a;
27    else if (i == SHL) a = *sp++ << a;
28    else if (i == SHR) a = *sp++ >> a;
29    else if (i == ADD) a = *sp++ +  a;
30    else if (i == SUB) a = *sp++ -  a;
31    else if (i == MUL) a = *sp++ *  a;
32    else if (i == DIV) a = *sp++ /  a;
33    else if (i == MOD) a = *sp++ %  a;
34
35    else if (i == OPEN) a = open((char *)sp[1], *sp);
36    else if (i == READ) a = read(sp[2], (char *)sp[1], *sp);
37    else if (i == CLOS) a = close(*sp);
38    else if (i == PRTF) { t = sp + pc[1]; a = printf((char *)t[-1], t[-2], t[-3], t[-4], t[-5], t[-6]); }
39    else if (i == MALC) a = (int)malloc(*sp);
40    else if (i == FREE) free((void *)*sp);
41    else if (i == MSET) a = (int)memset((char *)sp[2], sp[1], *sp);
42    else if (i == MCMP) a = memcmp((char *)sp[2], (char *)sp[1], *sp);
43    else if (i == EXIT) { printf("exit(%d) cycle = %d\n", *sp, cycle); return *sp; }
44    else { printf("unknown instruction = %d! cycle = %d\n", i, cycle); return -1; }
