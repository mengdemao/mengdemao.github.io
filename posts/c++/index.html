C++内存分区

c++程序在运行时,将内存分为4个区域

  1. 代码区: 存放程序的二进制代码,由操作系统管理
  2. 全局区: 存放全局变量、静态变量和常量
  3. 栈区: 编译器自动分配
  4. 堆区: 程序负责分配和释放

重载的原理:对已有的运算符进行重新定义,赋予新的功能含义;

 1class Person {
 2public:
 3    int m_A;
 4    int m_B;
 5
 6    /* 使用成员函数实现 */
 7    Person PersonAddPerson(Person &p)
 8    {
 9        Person t;
10        t.m_A = this->m_A + p.m_A;
11        t.m_B = this->m_B + p.m_B;
12        return t;
13    }
14
15    /* 重载+ */
16    Person operator+(Person &p)
17    {
18        Person t;
19        t.m_A = this->m_A + p.m_A;
20        t.m_B = this->m_B + p.m_B;
21        return t;
22    }
23};

递增重载++

注意:

  • 前置递增 p++
  • 后置递增 ++p

重载例子(复数)

 1#include <iostream>
 2
 3class Complex {
 4    friend std::ostream &operator<<(std::ostream &cout, Complex p);
 5
 6public:
 7    Complex(int i, int j);
 8    
 9    Complex();
10
11    /* 重载+ */
12    Complex operator+(Complex &p) 
13    {
14        Complex t;
15        t.i = this->i + p.i;
16        t.j = this->j + p.j;
17        return t;
18    }
19    /* 重载前置++ */
20    Complex& operator++()
21    {
22        this->i++;
23        this->j++;
24        return *this; 
25    }
26    
27    /* 重载后置++ */
28    Complex operator++(int)
29    {
30        Complex t;
31        
32        /* 记录 */
33        t.i = this->i;
34        t.j = this->j;
35
36        /* 递增 */
37        this->i++;
38        this->j++;
39
40        return t; 
41    }
42
43    /* 重载= */
44    Complex& operator=(Complex &p)
45    {
46        this->i = p.i;
47        this->j = p.j;
48
49        return *this;
50    }
51private:
52    int i;  /* 实部 */
53    int j;  /* 虚部 */
54};
55
56/* 构造函数 */
57Complex::Complex(int i, int j)
58{
59    this->i = i;
60    this->j = j;
61}
62
63Complex::Complex()
64{
65    this->i = 0;
66    this->j = 0;
67}
68
69std::ostream &operator<<(std::ostream &cout, Complex p)
70{
71    cout << p.i << "+" << p.j << "i"; 
72    return cout;
73} 
74
75int main(int argc, char *argv[])
76{
77    Complex p1(1, 2);
78    Complex p2(3, 4);
79
80    std::cout << p1 << std::endl;
81    std::cout << p2 << std::endl;
82    std::cout << p1 + p2 << std::endl;
83
84    std::cout << ++p1 << std::endl;
85    std::cout << p2++ << std::endl;
86
87    Complex p3 = p2 = p1;
88    std::cout << p1 << " " << p2 << " " << p3 << std::endl;
89
90    return 0;
91}

  • typename – 表明后面的符号是数据类型可以用class代替
