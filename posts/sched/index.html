 1static void __sched notrace __schedule(bool preempt)
 2{
 3	struct task_struct *prev, *next;
 4	unsigned long *switch_count;
 5	struct rq_flags rf;
 6	struct rq *rq;
 7	int cpu;
 8
 9	cpu = smp_processor_id();
10	rq = cpu_rq(cpu);
11	prev = rq->curr;
12
13	schedule_debug(prev);
14
15	if (sched_feat(HRTICK))
16		hrtick_clear(rq);
17
18	local_irq_disable();
19	rcu_note_context_switch(preempt);
20
21	rq_lock(rq, &rf);
22	smp_mb__after_spinlock();
23
24	rq->clock_update_flags <<= 1;
25	update_rq_clock(rq);
26
27	switch_count = &prev->nivcsw;
28	if (!preempt && prev->state) {
29		if (unlikely(signal_pending_state(prev->state, prev))) {
30			prev->state = TASK_RUNNING;
31		} else {
32			deactivate_task(rq, prev, DEQUEUE_SLEEP | DEQUEUE_NOCLOCK);
33			prev->on_rq = 0;
34
35			if (prev->in_iowait) {
36				atomic_inc(&rq->nr_iowait);
37				delayacct_blkio_start();
38			}
39			if (prev->flags & PF_WQ_WORKER) {
40				struct task_struct *to_wakeup;
41
42				to_wakeup = wq_worker_sleeping(prev);
43				if (to_wakeup)
44					try_to_wake_up_local(to_wakeup, &rf);
45			}
46		}
47		switch_count = &prev->nvcsw;
48	}
49
50	next = pick_next_task(rq, prev, &rf);
51	clear_tsk_need_resched(prev);
52	clear_preempt_need_resched();
53
54	if (likely(prev != next)) {
55		rq->nr_switches++;
56		rq->curr = next;
57		++*switch_count;
58
59		trace_sched_switch(preempt, prev, next);
60		rq = context_switch(rq, prev, next, &rf);
61	} else {
62		rq->clock_update_flags &= ~(RQCF_ACT_SKIP|RQCF_REQ_SKIP);
63		rq_unlock_irq(rq, &rf);
64	}
65
66	balance_callback(rq);
67}

寻找最高优先级任务

 1/*
 2 * Pick up the highest-prio task:
 3 */
 4static inline struct task_struct *
 5pick_next_task(struct rq *rq, struct task_struct *prev, struct rq_flags *rf)
 6{
 7	const struct sched_class *class;
 8	struct task_struct *p;
 9
10	/*
11	 * Optimization: we know that if all tasks are in the fair class we can
12	 * call that function directly, but only if the @prev task wasn't of a
13	 * higher scheduling class, because otherwise those loose the
14	 * opportunity to pull in more work from other CPUs.
15	 */
16	if (likely((prev->sched_class == &idle_sched_class ||
17		    prev->sched_class == &fair_sched_class) &&
18		   rq->nr_running == rq->cfs.h_nr_running)) {
19
20		p = fair_sched_class.pick_next_task(rq, prev, rf);
21		if (unlikely(p == RETRY_TASK))
22			goto again;
23
24		/* Assumes fair_sched_class->next == idle_sched_class */
25		if (unlikely(!p))
26			p = idle_sched_class.pick_next_task(rq, prev, rf);
27
28		return p;
29	}
30
31again:
32	for_each_class(class) {
33		p = class->pick_next_task(rq, prev, rf);
34		if (p) {
35			if (unlikely(p == RETRY_TASK))
36				goto again;
37			return p;
38		}
39	}
40
41	/* The idle class should always have a runnable task: */
42	BUG();
43}

任务切换

 1static __always_inline struct rq *
 2context_switch(struct rq *rq, struct task_struct *prev,
 3	       struct task_struct *next, struct rq_flags *rf)
 4{
 5	struct mm_struct *mm, *oldmm;
 6
 7	prepare_task_switch(rq, prev, next);
 8
 9	mm = next->mm;
10	oldmm = prev->active_mm;
11	/*
12	 * For paravirt, this is coupled with an exit in switch_to to
13	 * combine the page table reload and the switch backend into
14	 * one hypercall.
15	 */
16	arch_start_context_switch(prev);
17
18	/*
19	 * If mm is non-NULL, we pass through switch_mm(). If mm is
20	 * NULL, we will pass through mmdrop() in finish_task_switch().
21	 * Both of these contain the full memory barrier required by
22	 * membarrier after storing to rq->curr, before returning to
23	 * user-space.
24	 */
25	if (!mm) {
26		next->active_mm = oldmm;
27		mmgrab(oldmm);
28		enter_lazy_tlb(oldmm, next);
29	} else
30		switch_mm_irqs_off(oldmm, mm, next);
31
32	if (!prev->mm) {
33		prev->active_mm = NULL;
34		rq->prev_mm = oldmm;
35	}
36
37	rq->clock_update_flags &= ~(RQCF_ACT_SKIP|RQCF_REQ_SKIP);
38
39	prepare_lock_switch(rq, next, rf);
40
41	/* Here we just switch the register state and the stack. */
42	switch_to(prev, next, prev);
43	barrier();
44
45	return finish_task_switch(prev);
46}

 1void scheduler_tick(void)
 2{
 3	int cpu = smp_processor_id();
 4	struct rq *rq = cpu_rq(cpu);
 5	struct task_struct *curr = rq->curr;
 6	struct rq_flags rf;
 7
 8	sched_clock_tick();
 9
10	rq_lock(rq, &rf);
11
12	update_rq_clock(rq);
13	curr->sched_class->task_tick(rq, curr, 0);
14	cpu_load_update_active(rq);
15	calc_global_load_tick(rq);
16
17	rq_unlock(rq, &rf);
18
19	perf_event_task_tick();
20
21#ifdef CONFIG_SMP
22	rq->idle_balance = idle_cpu(cpu);
23	trigger_load_balance(rq);
24#endif
25}

优先级和运算时间计算

 1/*
 2 * Nice levels are multiplicative, with a gentle 10% change for every
 3 * nice level changed. I.e. when a CPU-bound task goes from nice 0 to
 4 * nice 1, it will get ~10% less CPU time than another CPU-bound task
 5 * that remained on nice 0.
 6 *
 7 * The "10% effect" is relative and cumulative: from _any_ nice level,
 8 * if you go up 1 level, it's -10% CPU usage, if you go down 1 level
 9 * it's +10% CPU usage. (to achieve that we use a multiplier of 1.25.
10 * If a task goes up by ~10% and another task goes down by ~10% then
11 * the relative distance between them is ~25%.)
12 */
13const int sched_prio_to_weight[40] = {
14 /* -20 */     88761,     71755,     56483,     46273,     36291,
15 /* -15 */     29154,     23254,     18705,     14949,     11916,
16 /* -10 */      9548,      7620,      6100,      4904,      3906,
17 /*  -5 */      3121,      2501,      1991,      1586,      1277,
18 /*   0 */      1024,       820,       655,       526,       423,
19 /*   5 */       335,       272,       215,       172,       137,
20 /*  10 */       110,        87,        70,        56,        45,
21 /*  15 */        36,        29,        23,        18,        15,
22};
23
24/*
25 * Inverse (2^32/x) values of the sched_prio_to_weight[] array, precalculated.
26 *
27 * In cases where the weight does not change often, we can use the
28 * precalculated inverse to speed up arithmetics by turning divisions
29 * into multiplications:
30 */
31const u32 sched_prio_to_wmult[40] = {
32 /* -20 */     48388,     59856,     76040,     92818,    118348,
33 /* -15 */    147320,    184698,    229616,    287308,    360437,
34 /* -10 */    449829,    563644,    704093,    875809,   1099582,
35 /*  -5 */   1376151,   1717300,   2157191,   2708050,   3363326,
36 /*   0 */   4194304,   5237765,   6557202,   8165337,  10153587,
37 /*   5 */  12820798,  15790321,  19976592,  24970740,  31350126,
38 /*  10 */  39045157,  49367440,  61356676,  76695844,  95443717,
39 /*  15 */ 119304647, 148102320, 186737708, 238609294, 286331153,
40};

42void reweight_task(struct task_struct *p, int prio)
43{
44	struct sched_entity *se = &p->se;
45	struct cfs_rq *cfs_rq = cfs_rq_of(se);
46	struct load_weight *load = &se->load;
47	unsigned long weight = scale_load(sched_prio_to_weight[prio]);
48
49	reweight_entity(cfs_rq, se, weight, weight);
50	load->inv_weight = sched_prio_to_wmult[prio];
51}
52
53static void set_load_weight(struct task_struct *p, bool update_load)
54{
55	int prio = p->static_prio - MAX_RT_PRIO;
56	struct load_weight *load = &p->se.load;
57
58	/*
59	 * SCHED_IDLE tasks get minimal weight:
60	 */
61	if (idle_policy(p->policy)) {
62		load->weight = scale_load(WEIGHT_IDLEPRIO);
63		load->inv_weight = WMULT_IDLEPRIO;
64		return;
65	}
66
67	/*
68	 * SCHED_OTHER tasks have to update their load when changing their
69	 * weight
70	 */
71	if (update_load && p->sched_class == &fair_sched_class) {
72		reweight_task(p, prio);
73	} else {
74		load->weight = scale_load(sched_prio_to_weight[prio]);
75		load->inv_weight = sched_prio_to_wmult[prio];
76	}
77}

任务队列

 1static inline void enqueue_task(struct rq *rq, struct task_struct *p, int flags)
 2{
 3	if (!(flags & ENQUEUE_NOCLOCK))
 4		update_rq_clock(rq);
 5
 6	if (!(flags & ENQUEUE_RESTORE))
 7		sched_info_queued(rq, p);
 8
 9	p->sched_class->enqueue_task(rq, p, flags);
10}
11
12static inline void dequeue_task(struct rq *rq, struct task_struct *p, int flags)
13{
14	if (!(flags & DEQUEUE_NOCLOCK))
15		update_rq_clock(rq);
16
17	if (!(flags & DEQUEUE_SAVE))
18		sched_info_dequeued(rq, p);
19
20	p->sched_class->dequeue_task(rq, p, flags);
21}
22
23void activate_task(struct rq *rq, struct task_struct *p, int flags)
24{
25	if (task_contributes_to_load(p))
26		rq->nr_uninterruptible--;
27
28	enqueue_task(rq, p, flags);
29}
30
31void deactivate_task(struct rq *rq, struct task_struct *p, int flags)
32{
33	if (task_contributes_to_load(p))
34		rq->nr_uninterruptible++;
35
36	dequeue_task(rq, p, flags);
37}
